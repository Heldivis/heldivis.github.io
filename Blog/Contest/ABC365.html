ABC365 G 题解

考虑对每个人开一个线段树，每次询问就是将两个线段树合并。

细节看代码：

void PushUp(int p) { st[p].dat = st[st[p].lc].dat + st[st[p].rc].dat; }
void Modify(int &p, int tl, int tr, int ql, int qr) {
if (tl > tr) return;
if (!p) p = ++cnt;
if (ql <= tl && tr <= qr) {
 st[p].dat = tr - tl + 1;
 return;
}
int mid = (tl + tr) >> 1;
if (ql <= mid) Modify(st[p].lc, tl, mid, ql, qr);
if (mid < qr) Modify(st[p].rc, mid + 1, tr, ql, qr);
PushUp(p);
}
ll Merge(int p, int q, int tl, int tr) {
if (st[p].dat == tr - tl + 1) return st[q].dat;
if (st[q].dat == tr - tl + 1) return st[p].dat;
if (!p || !q) return 0;
int mid = (tl + tr) >> 1;
return Merge(st[p].lc, st[q].lc, tl, mid) +
      Merge(st[p].rc, st[q].rc, mid + 1, tr);
}

非常简单，建议评红 @kkksc03 @chen_zhe。
