n = read();
for (int i = 1; i <= n; i++) a[i] = read();
for (int i = 1; i <= n; i++)
    for (int j = i + 1; j <= n; j++) ds.insert(a[j] - a[i]);
for (int i = 1; i <= n; i++) f[i][1] = 1;
for (int d : ds) {
    for (int i = 1; i <= n; i++)
        for (int k = 1; k < i; k++)
            if (a[i] - a[k] == d)
                for (int j = 1; j < i; j++) (f[i][j + 1] += f[k][j]) %= P;
    for (int i = 1; i <= n; i++)
        for (int j = 2; j <= n; j++) (res[j] += f[i][j]) %= P, f[i][j] = 0;
}
res[1] = n;
for (int i = 1; i <= n; i++) printf("%lld ", res[i]);
